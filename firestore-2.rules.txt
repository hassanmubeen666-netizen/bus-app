// ═══════════════════════════════════════════════════════════════════════
// Bustrip / Omnibus — Firestore Security Rules
// Project: bus-trip-sheet
// ═══════════════════════════════════════════════════════════════════════
//
// ⚠️  READ THIS FIRST — the honest limitation of these rules  ⚠️
//
// index.html and get-started.html do NOT use Firebase Authentication.
// Login is done by reading a `users/{username}` document from Firestore
// and comparing a hashed password *in the browser*. Firebase never issues
// that browser a real identity (`request.auth` stays null forever).
//
// That means these rules CANNOT tell "a logged-in admin" apart from
// "anyone on the internet with your Firebase config" — and your Firebase
// config (apiKey, projectId, etc.) is already sitting in plain text in
// your HTML source, which is normal for Firebase web apps but means it's
// not a secret. Someone could open dev tools, copy that config, and call
// the Firestore SDK directly — these rules are what stands between them
// and your data, NOT your login screen.
//
// What these rules DO give you:
//   - Structural validation (a "logs" write must look like a real log)
//   - Prevents wiping entire collections or writing garbage documents
//   - Prevents a stranger from silently editing someone else's password
//     hash to hijack an account
//   - Keeps other people's raw data un-readable in a few sensitive spots
//
// What these rules CANNOT give you (without real auth):
//   - "Only admins can see the Reports tab" — enforced by role at the
//     database level. Right now that's a UI-only restriction in the app.
//   - Rate limiting login attempts
//   - Truly private per-fleet data if you ever go multi-tenant
//
// The real fix, when you're ready: migrate login to Firebase Authentication
// (e.g. a Cloud Function that checks the Firestore password hash once,
// then mints a custom token so the client signs in for real via
// signInWithCustomToken). Ask me to help set that up — it's a bigger
// change than these rules, but it's the only way to make role checks
// actually enforceable server-side. Everything below is the best
// practical middle ground until then.
//
// ⚠️  MULTI-TENANCY MAKES THIS MORE IMPORTANT, NOT LESS  ⚠️
// Now that different companies (Mannan Travels, and whoever signs up
// next) share this same Firestore project, "isolated per company" is
// currently a CLIENT-SIDE filter only — the app fetches every company's
// logs/staff/vehicles/etc. and hides the ones that don't match your
// companyId. These rules do not check companyId ownership, because
// there's no real identity to check it against (see above). Practically:
// anyone who opens dev tools and calls the Firestore SDK directly could
// read or write another company's trip data right now. This is the
// single strongest argument for doing the Firebase Auth migration before
// you have paying customers who'd mind that.
//
// get-started.html also has a honeypot field, a 3-second minimum fill
// time, and a per-tab cooldown to cut down on casual signup spam. All of
// that is client-side JavaScript, easily bypassed by anyone calling the
// Firestore SDK directly — it filters out lazy bots, not a targeted one.
// It does NOT replace anything these rules should be doing.
// ═══════════════════════════════════════════════════════════════════════

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // ── users/{username} ──────────────────────────────────────────────
    // Doc id = username. Read is required for login/signup checks (the
    // app has to look up the doc to verify a password) — this is the
    // clearest example of the auth gap above: this collection holds
    // password hashes and must be readable by the app pre-login, so it's
    // effectively public-readable. Hashing (already in place) is what
    // protects the actual passwords if this is scraped.
    //
    // Multi-tenant note: every user belongs to a companyId. New company
    // signups set companyId to their own username (they're the company's
    // first admin); staff/drivers added later inherit their admin's
    // companyId. Docs with no companyId field are treated by the app as
    // belonging to the original "mannan-travels" account.
    match /users/{username} {
      allow get: if true;
      allow list: if false; // no reason to ever list all usernames

      allow create: if request.resource.data.username == username
                    && request.resource.data.role in ["admin","staff","driver"]
                    && request.resource.data.password is string
                    && request.resource.data.password.matches("^sha256:.*")
                    && request.resource.data.active is bool
                    && request.resource.data.displayName is string
                    && request.resource.data.displayName.size() > 0;

      // updates are allowed (password changes, activate/deactivate,
      // upgrading legacy plaintext passwords, approving pending signups)
      // but can't change the username or hand yourself a different role
      allow update: if request.resource.data.username == resource.data.username
                    && (!("role" in request.resource.data)
                        || request.resource.data.role == resource.data.role
                        || request.resource.data.role in ["admin","staff","driver"]);

      allow delete: if true; // matches current in-app "delete/reject user" feature
    }

    // ── logs/{logId} — daily trip P&L entries ───────────────────────────
    match /logs/{logId} {
      allow read: if true;
      allow create, update: if request.resource.data.date is string
                    && request.resource.data.date.matches("^[0-9]{4}-[0-9]{2}-[0-9]{2}$")
                    && request.resource.data.revenue is number
                    && request.resource.data.expenses is number
                    && request.resource.data.profit is number;
      allow delete: if true;
    }

    // ── staff/{staffId} — driver/cleaner register ───────────────────────
    match /staff/{staffId} {
      allow read: if true;
      allow create, update: if request.resource.data.name is string
                    && request.resource.data.name.size() > 0;
      allow delete: if true;
    }

    // ── vehicles/{vehicleId} — fleet register & document expiry ────────
    match /vehicles/{vehicleId} {
      allow read: if true;
      allow create, update: if request.resource.data.vehicle_number is string
                    && request.resource.data.vehicle_number.size() > 0;
      allow delete: if true;
    }

    // ── tax/{taxId} & emi/{emiId} — due-date reminders ──────────────────
    match /tax/{taxId} {
      allow read: if true;
      allow create, update: if request.resource.data.amount is number
                    && request.resource.data.due_date is string;
      allow delete: if true;
    }
    match /emi/{emiId} {
      allow read: if true;
      allow create, update: if request.resource.data.amount is number
                    && request.resource.data.due_date is string;
      allow delete: if true;
    }

    // ── access_log/{entryId} — login/security audit trail ───────────────
    // Write-mostly: the app logs every login attempt, including failed
    // and pre-login ones, so create must stay open. Reading this back
    // (device/browser info per attempt) is only meant for admins inside
    // the app, but — per the caveat above — that's a UI convention only.
    // If this ever needs to be truly private, it needs real auth.
    match /access_log/{entryId} {
      allow read: if true;
      allow create: if request.resource.data.username is string
                    && request.resource.data.status in ["success","failed"]
                    && request.resource.data.timestamp is number;
      allow update, delete: if false; // audit trail — never edited or deleted by the app
    }

    // ── everything else: closed by default ──────────────────────────────
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
